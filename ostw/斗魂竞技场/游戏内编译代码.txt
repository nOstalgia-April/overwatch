rule("Execute Next Async Operation")
{
	event
	{
		Ongoing - Each Player;
		All;
		All;
	}

	conditions
	{
		Total Time Elapsed >= Event Player.timeUntilNextOperation;
	}

	actions
	{
		Event Player.opIndex = First Of(Sorted Array(Mapped Array(Event Player.operations, Current Array Index), First Of(
			Event Player.operations[Current Array Element])));
		Modify Global Variable(func_group, Append To Array, Array(Event Player.operations[Event Player.opIndex][1]));
		Call Subroutine(func_group);
		Modify Player Variable(Event Player, operations, Remove From Array By Index, Event Player.opIndex);
		Event Player.timeUntilNextOperation = Count Of(Event Player.operations) ? First Of(First Of(Sorted Array(Event Player.operations,
			First Of(Current Array Element)))) : 99999;
		Wait(0.016, Ignore Condition);
		Loop If Condition Is True;
	}
}

				Global._extendedGlobalCollection[112] = Global._arrayConstructor;
				Wait(0.016, Ignore Condition);
			End;
			Modify Player Variable(Event Player, roundReward, Append To Array, Array(
				Event Player.roundReward[Event Player.roundReward_index]));
			Event Player.roundReward[Event Player.roundReward_index] = Last Of(Global._extendedGlobalCollection[112]);
			Event Player.RefreshCount -= 1;
			Play Effect(Event Player, Soldier: 76 Sprint Start Sound, Color(White), Eye Position(Event Player) + Facing Direction Of(
				Event Player), 200);
			Modify Global Variable At Index(_extendedGlobalCollection, 112, Remove From Array By Index, Count Of(
				Global._extendedGlobalCollection[112]) - 1);
		End;
		Skip(48);
		Event Player.forgeReward_index = 0;
		Skip(46);
		Event Player.forgeReward_index = 1;
		Skip(44);
		Event Player.forgeReward_index = 2;
		Skip(42);
		Global._extendedGlobalCollection[47] = Event Player;
		Call Subroutine(OnClickYesofRewards);
		Skip(39);
		If(Event Player.RefreshCount > 0);
			Modify Global Variable At Index(_extendedGlobalCollection, 113, Append To Array, Array(
				Global._objectVariable_0[Event Player.forgeReward[Event Player.forgeReward_index]]));
			Modify Global Variable At Index(_extendedGlobalCollection, 114, Append To Array, Array(0));
			If(Last Of(Global._extendedGlobalCollection[113]) == 0);
				Call Subroutine(GenerateAverageForge);
				Global._arrayConstructor = Global._extendedGlobalCollection[114];
				Global._arrayConstructor[Count Of(Global._extendedGlobalCollection[114]) - 1] = Global.returnValue_GenerateAverageForge;
				Global._extendedGlobalCollection[114] = Global._arrayConstructor;
				While(Array Contains(Event Player.forgeReward, Last Of(Global._extendedGlobalCollection[114])) || Array Contains(
					Event Player.Forges, Last Of(Global._extendedGlobalCollection[114])));
					Call Subroutine(GenerateAverageForge);
					Global._arrayConstructor = Global._extendedGlobalCollection[114];
					Global._arrayConstructor[Count Of(Global._extendedGlobalCollection[114]) - 1] = Global.returnValue_GenerateAverageForge;
					Global._extendedGlobalCollection[114] = Global._arrayConstructor;
				End;
			Else;
				Global._arrayConstructor = Global._extendedGlobalCollection[114];
				Global._arrayConstructor[Count Of(Global._extendedGlobalCollection[114]) - 1] = Random Value In Array(Filtered Array(
					Global.RareForgePool, !Array Contains(Event Player.forgeReward, Current Array Element) && !Array Contains(Event Player.Forges,
					Current Array Element)));
				Global._extendedGlobalCollection[114] = Global._arrayConstructor;
			End;
			Modify Player Variable(Event Player, forgeReward, Append To Array, Event Player.forgeReward[Event Player.forgeReward_index]);
			Event Player.forgeReward[Event Player.forgeReward_index] = Last Of(Global._extendedGlobalCollection[114]);
			Event Player.RefreshCount -= 1;
			Play Effect(Event Player, Tracer Recall Reappear Sound, Color(White), Eye Position(Event Player) + Facing Direction Of(
				Event Player), 150);
			Modify Global Variable At Index(_extendedGlobalCollection, 113, Remove From Array By Index, Count Of(
				Global._extendedGlobalCollection[113]) - 1);
			Modify Global Variable At Index(_extendedGlobalCollection, 114, Remove From Array By Index, Count Of(
				Global._extendedGlobalCollection[114]) - 1);
		End;
		Skip(12);
		Event Player.roundReward = Empty Array;
		If(Global.CurrentRound != 7);
			Global.p_4 = Event Player;
			Global.isRare = False;
			Call Subroutine(GenerateForgeRewards);
		Else;
			Global.p_4 = Event Player;
			Global.isRare = True;
			Call Subroutine(GenerateForgeRewards);
		End;
		Event Player.menuactive = 2;
		Play Effect(Event Player, Soldier: 76 Sprint Start Sound, Color(White), Eye Position(Event Player), 200);
		Modify Global Variable(func_group, Remove From Array By Index, Count Of(Global.func_group) - 1);
	}
